const numbers = [1, 2, 3, 4, 5];
const squares = numbers.map(square);

console.log(squares);

function square(element){
    return Math.pow(element, 2)
}

O método `map` é usado para criar um novo array `squares`, aplicando a função `square` a cada elemento do array `numbers`.
